# **1️⃣ Main Entrance (IoT9)**

```javascript
// Allowed RFID cards for Main Entrance
var allowedCards = ["RFID_Security", "RFID_LabTech", "RFID_Lecturer", "RFID_Student", "RFID_Visitor"];

function unlockDoor(cardName) {
    if (allowedCards.includes(cardName)) {
        syslog.log("Access granted for " + cardName + " at Main Entrance (IoT9) on " + new Date().toLocaleString());
        device.doorLock = false;
        device.sendMessage("SecurityPC", cardName + " accessed Main Entrance (IoT9)");
    } else {
        syslog.log("Unauthorized access attempt by " + cardName + " at Main Entrance (IoT9) on " + new Date().toLocaleString());
        device.doorLock = true;
        device.sendMessage("SecurityPC", "Unauthorized access attempt by " + cardName + " at Main Entrance (IoT9)");
    }
}

device.onCardScanned = function(card) {
    unlockDoor(card.name);
};
```
